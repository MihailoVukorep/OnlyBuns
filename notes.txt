# create user & database
sudo -iu postgres
psql
CREATE USER postgres WITH PASSWORD 'root'; 
CREATE DATABASE jpa OWNER postgres; 
GRANT ALL PRIVILEGES ON DATABASE jpa TO postgres; 
\du
